function carregarAlunos(){
    const tid = document.getElementById('sel-turma').value;
    const selAluno = document.getElementById('sel-aluno');
    
    if(!tid){
        selAluno.innerHTML = '<option value="">Aguardando turma...</option>';
        selAluno.disabled = true;
        return;
    }
    
    const t = turmasPublicas.find(x => x.id === tid);
    if(t && t.alunos && t.alunos.length > 0){
        const ordenados = [...t.alunos].sort((a,b)=>a.nome.localeCompare(b.nome));
        selAluno.innerHTML = '<option value="">Selecione o seu nome...</option>' + ordenados.map(a => `<option value="${a.id}">${a.nome}</option>`).join('');

    }

// Interação do Checklist Local
function toggleChecklist(idCheck, elemento){
    const key = 'aluno_checks_' + ALUNO.id;
    let progresso = JSON.parse(localStorage.getItem(key) || '{}');
    
    if(progresso[idCheck]){
        delete progresso[idCheck];
        elemento.classList.remove('feito');
        elemento.querySelector('.chk-box').innerHTML = '';
    } else {
        progresso[idCheck] = true;
        elemento.classList.add('feito');
        elemento.querySelector('.chk-box').innerHTML = '✓';
    }
    localStorage.setItem(key, JSON.stringify(progresso));
}

// --- FLUXO 3: RESPONDER A PROVA ---
function iniciarProva(id){
    provaAtual = provasPublicadas.find(x => x.id === id);
    if(!provaAtual) return;
    
    document.getElementById('prova-titulo').textContent = provaAtual.titulo;
    const container = document.getElementById('prova-questoes');
    
    container.innerHTML = provaAtual.perguntas.map((q, i) => {
        let html = `<div class="q-box" id="qbox-${i}">
            <div class="q-enum">${i+1}. ${q.enunciado}</div>`;
            
        if(q.tipo === 'multipla'){
            html += `<div class="opcoes-container" data-qidx="${i}">`;
            q.alternativas.forEach((alt, altIdx) => {
                html += `
                <label class="alt-label">
                    <input type="radio" name="resp_${i}" value="${altIdx}"/>
                    <span style="font-size:15px;line-height:1.4;color:var(--text);margin-top:2px">${alt}</span>
                </label>`;
            });
            html += `</div>`;
        } else {
            html += `<textarea id="resp_${i}" placeholder="Escreva a sua resposta de forma clara..."></textarea>`;
        }
        html += `</div>`;
        return html;
    }).join('');
    
    document.getElementById('btn-enviar').disabled = false;
    document.getElementById('btn-enviar').innerHTML = 'Entregar Avaliação ✅';
    mostrarTela('tela-prova');
    window.scrollTo(0,0);
}

async function enviarProva(){
    // 1. Validação
    let todasRespondidas = true;
    let acertosME = 0;
    let totalME = 0;
    let temasOk = [];
    let temasFragos = [];
    let dissertativas = [];
    let mcLog = [];

    provaAtual.perguntas.forEach((q, i) => {
        if(q.tipo === 'multipla'){
            totalME++;
            const radios = document.getElementsByName(`resp_${i}`);
            let respondida = false; let val = null; let textoResp = '';
            for(let r of radios){ if(r.checked){ respondida = true; val = parseInt(r.value); textoResp = q.alternativas[val]; break; } }
            
            if(!respondida) todasRespondidas = false;
            
            const isCorrect = (val === q.gabarito);
            if(isCorrect){ acertosME++; temasOk.push(q.enunciado.substring(0, 30) + '...'); } 
            else { temasFragos.push(q.enunciado.substring(0, 30) + '...'); }
            
            mcLog.push({ enunciado: q.enunciado, respostaAluno: textoResp || 'Em branco', correto: isCorrect });
        } else {
            const txtArea = document.getElementById(`resp_${i}`);
            const val = txtArea.value.trim();
            if(!val) todasRespondidas = false;
            dissertativas.push({ enunciado: q.enunciado, respostaAluno: val || '' });
        }
    });

    if(!todasRespondidas){
        alert('Por favor, responda a todas as questões antes de entregar.');
        return;
    }

    // 2. Payload de submissão
    const placar = totalME > 0 ? Math.round((acertosME / totalME) * 100) : 100;
    
    const payload = {
        provaId: provaAtual.id,
        tituloProva: provaAtual.titulo,
        alunoId: ALUNO.id,
        nome: ALUNO.nome,
        turma: ALUNO.turmaNome,
        disciplina: ALUNO.disciplina,
        placar: placar,
        acertos: acertosME,
        total: totalME,
        temasOk: temasOk,
        temasFragos: temasFragos,
        respostasMC: mcLog,
        dissertativas: dissertativas,
        dataHora: new Date().toISOString()
    };

    // 3. Enviar
    document.getElementById('btn-enviar').disabled = true;
    document.getElementById('btn-enviar').innerHTML = '<div class="spinner" style="width:20px;height:20px;margin:0;border-width:2px"></div> Enviando...';

    try {
        const {collection, addDoc} = await import('[https://www.gstatic.com/firebasejs/10.12.0/firebase-firestore.js](https://www.gstatic.com/firebasejs/10.12.0/firebase-firestore.js)');
        await addDoc(collection(_db, 'quiz_resultados'), payload);
        mostrarTela('tela-sucesso');
        window.scrollTo(0,0);
    } catch(e) {
        alert('Erro de conexão ao enviar a prova. Verifique sua internet.');
        document.getElementById('btn-enviar').disabled = false;
        document.getElementById('btn-enviar').innerHTML = 'Entregar Avaliação ✅';
    }
}

    document.getElementById(id).classList.add('act');
}
